import PropTypes from "prop-types"
import React from 'react';

/**
 * Welcome
 * @param {*} param0 
 * @public
 * @returns {JSX}
 */


const Welcome = ({datas}) => {

    return (
        <div className='welcome'>
            <h1>Bonjour <span>{datas.user?.userInfos.firstName}</span></h1>

}

export default Welcome;
